    $("img").each(
        function () {
            let img = $(this);
            let index = parseInt(img.index());
            console.log(index);
            let dis = 130;
            let zindex = 1;

            dis = dis * index;
            zindex = zindex + index;

            console.log(dis)
            console.log("------")
            img.css({
                "left": dis,
                // 'z-index':zindex
            })
        }
    ).click(
        function () {
            $("img").animate({
                "margin-top": 0,
                'z-index': $(this).index(),
                "width": "323px",
                "height": "208px"
            })
            $(this).animate({
                "margin-top": 100,
                'z-index': 5,
                "width": "646px",
                "height": "416px"
            });
        }
    ).mouseover(function () {
        $(this).stop(true,true).animate({
            "margin-left": 20
        }).animate({
            "margin-left": 0
        });
    }).mouseout(function () {
        $(this).animate({
            "margin-left": 0
        });
    });
