jQuery实现div的放大
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="../jquery-3.6.3.js"></script>
    <style>
        div{
            width: 100px;
            height: 100px;
            background-color: orange;
            position: relative;
        }
    </style>
</head>
<body>
<!--css3 transition 动画-->
<div id="d1">

</div>

<script>
    jQuery("#d1").stop(true).mouseover(function (){
        this.animate({
            'width':"300px",
            'height':"300px"
        },1000);
    });
</script>
</body>
</html>
